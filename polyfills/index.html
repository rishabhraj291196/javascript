Polyfills

Filter polyfills


//normal way
const arr = [1,2,3,4];
const modified = arr.filter((val) => val > 2);
console.log(modified);
//output : [3,4]

//polyfills filter
Array.prototype.myFilter = function(cb){
	const temp = [];
	for(let i = 0; i < this.length; i++){
		if(cb(this[i],i,this)) temp.push(this[i])
	}
	return temp;
}

const modified2 = arr.myFilter((val) => val > 2);
console.log(modified2);
//output : [3,4]

Map polyfills


//normal way
const arr = [1,2,3,4];
const modified = arr.map((val) => val * 2);
console.log(modified);
//output : [2,4,6,8]

//polyfills filter
Array.prototype.myMap = function(cb){
	const temp = [];
	for(let i = 0; i < this.length; i++){
		temp.push(this[i] * 2)
	}
	return temp;
}

const modified2 = arr.myMap((val) => val * 2);
console.log(modified2);
//output : [2,4,6,8]
